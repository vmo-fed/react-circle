# react-circle

### 安装

```shell
npm i @vmo-fed/react-circle -S
```

### 使用

```jsx
import Circle from 'react-circle';

class App extends React.Component {
  render() {
    return (
      <Circle progress={90} strokeColor={['#05a', '#0a5']} />
    );
  }
}
```
